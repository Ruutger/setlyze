[docs]def text(key, *args):
    """Return the full text string from the dictionary."""
    if key not in ENGLISH:
        logging.error("setlyze.locale.text: unknown key '%s'" % key)
        sys.exit(1)

    if args:
        text = ENGLISH[key] % (args)
    else:
        text = ENGLISH[key]

    return text
